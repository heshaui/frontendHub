 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
    if (x === null || x === undefined || x === '') return false;
    x += ""
    const len = x.length
    if (len === 1) return true;
    if (len === 2) return x[0] === x[1];
    let isEven = len % 2 === 0;
    let midIndex = Math.floor(len / 2); // ä¸­é—´çš„ç´¢å¼•
    let isPalind = function (str, left, right) {
        let l = left,
            r = right;
        while (l >= 0 && r <= str.length - 1) {
            if (str[l] === str[r]) {
                l--;
                r++;
            } else {
                return false;
            }
        }
        return true;
    }
    return isEven ? isPalind(x, midIndex - 1, midIndex) : isPalind(x, midIndex - 1, midIndex + 1)
};

var isPalindrome = function(x) {
    if (typeof x !== 'number') return false;
    let str = String(x);
    let i = 0, j = str.length - 1;
    while (i < j) {
        if(str.charAt(i) !== str.charAt(j)) return false;
        i++;
        j--;
    }
    return true;
